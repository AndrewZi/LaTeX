    \begin{enumerate} [label=\Alph*] % \Alph - заглавные буквы латинского алфавита

        \setcounter{enumi}{5}
        \item another item % Здесь счётчик enumi = 6
    \end{enumerate}
    

    С римскими числами:

    \end{enumerate}
