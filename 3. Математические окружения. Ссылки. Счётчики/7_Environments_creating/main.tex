    % name - название создаваемого окружения

    % succeeding_code - код после объекта, на который воздействует окружение

    \newenvironment{centerbold} % centerbold - название создаваемого окружения
    % Не выдаёт ошибку компиляции (например, незакрытое окружение) до использования
    {\begin{center}\bfseries} % <-- preceding_code
% \bfseries - модификатор (работает до конца блока, где добавлен, т. е. окружения center)
    {\end{center}} % <-- succeeding_code

    \begin{centerbold}
        Текст.
    \end{centerbold}

    % Что делает LaTeX:
    \begin{center}\bfseries % <-- preceding_code
% \bfseries - модификатор (работает до конца блока, где добавлен, т. е. окружения center)
        Текст.
    \end{center} % <-- succeeding_code
